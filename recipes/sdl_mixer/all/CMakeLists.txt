project(sdl2_mixer)

include(CheckTypeSize)

macro(add_music_option type)
  option(${type} "${type} music support" OFF)
  message(STATUS "${type} ${${type}}")
  if(${${type}})
    add_definitions("-DMUSIC_${type}")
    if(${${type}_DYNAMIC})
      add_definitions("-D${type}_DYNAMIC")
    endif()
  endif()
endmacro()

add_music_option(CMD)
